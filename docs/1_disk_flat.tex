\section{Flat-membrane, tilt-only benchmark (inner $I_\nu$ patch + outer $K_\nu$ decay)}
\label{sec:tilt_only_flat_two_region}

This section presents the simplest analytically solvable control case in which
the membrane midplane is \emph{not allowed to bend} (flat membrane),
but a lipid-tilt field is generated by a single caveolin disk.
The tilt exists both in the patch above the disk ($0\le r < R$) and in the distal
membrane ($r>R$), where it decays to zero at infinity.

\subsection{Assumptions}

\begin{itemize}
\item The membrane height is fixed: $z(r)\equiv 0$ (no bending degree of freedom).
\item The only field is the scalar tilt angle $\theta(r)$ (axisymmetric).
\item A disk of radius $R$ imposes a boundary tilt $\theta(R)=\theta_B$.
\item Regularity at the origin: $\theta(r)$ remains finite as $r\to 0$.
\item Far-field decay: $\theta(r)\to 0$ as $r\to\infty$.
\end{itemize}

\subsection{Energy functional and definition of $\lambda$}

We use the standard quadratic ``massive'' tilt functional
\begin{equation}
F_{\rm tilt}
=
\frac{\kappa_t}{2}\int \left[(\nabla \theta)^2 + \lambda^{-2} \theta^2\right]\, dA ,
\end{equation}
where $\kappa_t$ is the tilt modulus and $\lambda^{-1}$ is the intrinsic decay
length of the tilt field.

We adopt the convention
\begin{equation}
\boxed{\lambda^{-2}=\frac{\kappa_t}{\kappa}}
\qquad \Longleftrightarrow \qquad
\boxed{\lambda=\sqrt{\frac{\kappa}{\kappa_t}}}.
\end{equation}

In axisymmetry this becomes, split into inner ($0\le r\le R$) and outer ($R\le r<\infty$) regions,
\begin{equation}
F_{\rm tilt}
=
\frac{\kappa_t}{2}\left[
\int_{0}^{R}\left(\theta'^2+\lambda^{-2}\theta^2\right)2\pi r\,dr
+
\int_{R}^{\infty}\left(\theta'^2+\lambda^{-2}\theta^2\right)2\pi r\,dr
\right].
\end{equation}

The rim contact term is taken in the linear form
\begin{equation}
F_{\rm contact}
=
-2\pi R\left(\frac{h\Delta\varepsilon}{a}\right)\theta_B,
\end{equation}
so that the total energy is
\begin{equation}
F_{\rm tot}(\theta_B)=F_{\rm tilt}(\theta_B)+F_{\rm contact}(\theta_B).
\end{equation}

\subsection{Euler--Lagrange equation and solutions}

Variation of $F_{\rm tilt}$ gives
\begin{equation}
\theta''+\frac{1}{r}\theta' - \lambda^{-2}\theta = 0.
\end{equation}

The unique axisymmetric solutions satisfying the boundary conditions are:

\paragraph{Inner (above-disk) patch, $0\le r\le R$:}
regularity at $r=0$ selects $I_0$,
\begin{equation}
\theta_{\rm in}(r)=\theta_B\frac{I_0(r/\lambda)}{I_0(R/\lambda)}.
\end{equation}

\paragraph{Outer (distal) membrane, $r\ge R$:}
decay at infinity selects $K_0$,
\begin{equation}
\theta_{\rm out}(r)
=
\theta_B\frac{K_0(r/\lambda)}{K_0(R/\lambda)}.
\end{equation}

\subsection{Elastic energy as a boundary term}

Using the identity (valid for solutions of the Euler--Lagrange equation)
\begin{equation}
\frac{d}{dr}\left(r\,\theta\,\theta'\right)
=
r\left(\theta'^2+\lambda^{-2}\theta^2\right),
\end{equation}
each region's elastic energy reduces to a boundary term:
\begin{equation}
F_{\rm tilt}^{[r_1,r_2]}
=
\pi\kappa_t\left[r\,\theta\,\theta'\right]_{r_1}^{r_2}.
\end{equation}

\paragraph{Inner elastic energy ($0\to R$):}
since $r\theta\theta'\to 0$ at $r\to 0$,
\begin{equation}
F_{\rm in,el}
=
\pi\kappa_t R\,\theta_B\,\theta'_{\rm in}(R).
\end{equation}
With $\theta'_{\rm in}(r)=\theta_B\lambda^{-1}\,I_1(r/\lambda)/I_0(R/\lambda)$,
\begin{equation}
F_{\rm in,el}
=
\pi\kappa_t R\,\lambda^{-1}\,\theta_B^2\,
\frac{I_1(R/\lambda)}{I_0(R/\lambda)}.
\end{equation}

\paragraph{Outer elastic energy ($R\to\infty$):}
since $r\theta\theta'\to 0$ at $r\to\infty$,
\begin{equation}
F_{\rm out,el}
=
-\pi\kappa_t R\,\theta_B\,\theta'_{\rm out}(R).
\end{equation}
With $\theta'_{\rm out}(r)= -\theta_B\lambda^{-1}\,K_1(r/\lambda)/K_0(R/\lambda)$,
\begin{equation}
F_{\rm out,el}
=
\pi\kappa_t R\,\lambda^{-1}\,\theta_B^2\,
\frac{K_1(R/\lambda)}{K_0(R/\lambda)}.
\end{equation}

Therefore the total elastic contribution is
\begin{equation}
F_{\rm el}(\theta_B)
=
F_{\rm in,el}+F_{\rm out,el}
=
\pi\kappa_t R\,\lambda^{-1}\,\theta_B^2
\left[
\frac{I_1(R/\lambda)}{I_0(R/\lambda)}
+
\frac{K_1(R/\lambda)}{K_0(R/\lambda)}
\right].
\end{equation}

\subsection{Total energy, optimum, and minimal value}

Collecting terms,
\begin{equation}
F_{\rm tot}(\theta_B)=A\,\theta_B^2 - B\,\theta_B,
\end{equation}
with
\begin{align}
A &= \pi\kappa_t R\,\lambda^{-1}
\left[
\frac{I_1(R/\lambda)}{I_0(R/\lambda)}
+
\frac{K_1(R/\lambda)}{K_0(R/\lambda)}
\right],
\\
B &= 2\pi R\left(\frac{h\Delta\varepsilon}{a}\right).
\end{align}

Minimization gives
\begin{equation}
\theta_B^\ast=\frac{B}{2A},
\qquad
F_{\rm tot}^\ast=-\frac{B^2}{4A}.
\end{equation}

\subsection{Numerical evaluation for the simulation parameter set}

Using the simulation values
\begin{equation}
\kappa = 1, \qquad
\kappa_t = 225,
\end{equation}
the definition
\begin{equation}
\lambda^{-2} = \frac{\kappa_t}{\kappa}
\end{equation}
gives
\begin{equation}
\lambda = \sqrt{\frac{\kappa}{\kappa_t}} = \frac{1}{15},
\qquad
\lambda^{-1} = \sqrt{\frac{\kappa_t}{\kappa}} = 15.
\end{equation}
Therefore,
\begin{equation}
\frac{R}{\lambda} = R\,\lambda^{-1}
= 0.4666666667 \times 15 = 7.
\end{equation}

At $R/\lambda = 7$,
\begin{equation}
\frac{I_1(7)}{I_0(7)} \simeq 0.9255322,
\qquad
\frac{K_1(7)}{K_0(7)} \simeq 1.0691785.
\end{equation}

Therefore
\begin{equation}
A \simeq 9869.8456,
\qquad
B = 2\pi R\left(\frac{h\Delta\varepsilon}{a}\right)\simeq 12.5663706.
\end{equation}

The optimal boundary tilt and minimal total energy are
\begin{equation}
\theta_B^\ast \simeq 6.366\times 10^{-4},
\qquad
F_{\rm tot}^\ast \simeq -3.9999\times 10^{-3},
\end{equation}
(in units of $\kappa$).

\paragraph{Energy breakdown at the optimum.}
At $\theta_B^\ast$,
\begin{align}
F_{\rm in,el} &\simeq +1.8559\times 10^{-3},\\
F_{\rm out,el} &\simeq +2.1440\times 10^{-3},\\
F_{\rm contact} &\simeq -7.9998\times 10^{-3},
\end{align}
so that
\begin{equation}
F_{\rm tot}^\ast
=
F_{\rm in,el}+F_{\rm out,el}+F_{\rm contact}
\simeq -3.9999\times 10^{-3}.
\end{equation}

\noindent
This flat-membrane tilt-only benchmark isolates tilt relaxation physics
(inner regular $I_0$ patch and outer decaying $K_0$ membrane) from bending effects.
